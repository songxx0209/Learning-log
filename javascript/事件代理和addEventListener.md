    window.onload = function() {
        var ulNode = document.getElementById("list");
        ulNode.addEventListener('click', function(e) {
            console.log(e);
            if (e.target && e.target.nodeName.toUpperCase() == "LI") { /*判断目标事件是否为li*/
                alert(e.target.innerHTML);
            }
        }, true);
    };
    /*
